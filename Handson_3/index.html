Q1-Find the number of digits

const Find_Digits = (N) => 
{
  let ans =0;
  while(N!==0){
    ans++;
    N=Math.floor(N/10);
  }
  return ans;
	 
};

Q3-Find Sum

const findSum = (n) => {
    let sum =0;
    for(i=1;i<=n;i++){
      if(i%2===0){
        sum =sum+i;
      }
    }
    return sum;
};
